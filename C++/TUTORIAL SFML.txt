#include <SFML/Graphics.hpp>
#include <fstream>
#include <iostream>
#include <windows.h>

using namespace std;
using namespace sf;

int main()
{
  sf::RenderWindow window(sf::VideoMode(640, 640), "Test program by Floodie");

  while (window.isOpen())
    {

      sf::Event event;
      while (window.pollEvent(event))
        {
	  if (event.type == sf::Event::Closed)
	    window.close();
        }
      window.clear();
      window.display();
    }

  return 0;
}
